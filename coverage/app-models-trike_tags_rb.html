297   def sibling_page(flag, tag)
298     page_index = case flag
299                  when :next
300                    1
301                  when :previous
302                    -1
303                  else
304                    raise ArgumentError, "flag must be :next or :previous"
305                  end
306     current = tag.locals.page
307     order_by = Page.column_names.include?('position') ? 'position' : 'title'
308     by = (tag.attr['by'] || order_by).strip
309 
310     unless current.attributes.keys.include?(by)
311       raise StandardTags::TagError.new("`by' attribute of `#{flag}' tag must be set to a valid page attribute name.")
312     end
313     # get the page's siblings, exclude any that have nil for the sorting
314     # attribute, exclude virtual pages and unpublished pages, and sort by the chosen attribute
315     siblings = current.self_and_siblings.delete_if { |s| s.send(by).nil? || s.virtual? || !s.published? }.sort_by { |page| page.attributes[by] }
316     if index = siblings.index(current)
317       new_page_index = index + page_index
318       new_page = new_page_index >= 0 ? siblings[new_page_index] : nil
319 
320       if new_page
321         tag.locals.page = new_page
322         tag.expand
323       end
324     end
325   end

327   def host(tag)
328     host = nil
329     page = tag.locals.page
330     root = page.root
331     if (host_part = root.part('host'))
332       host = host_part.content.sub(%r{/?$},'').sub(%r{^https?://},'').strip # strip trailing slash or leading protocol
333     elsif root.respond_to?(:site) && root.site
334       # multi_site extension is running
335       host = root.site.base_domain
336     elsif (request = tag.globals.page.request) && request.host
337       host = request.host
338       if host.nil? || host.empty? || host.match(/^\s*$/)
339         raise(StandardTags::TagError.new("request.host is returning something very unexpected (#{request.host.inspect}). You could override this behaviour by providing a 'host' page part on the site root page that contains the hostname."))
340       end
341     end
342     if host.nil? || host.empty? || host.match(/^\s*$/)
343       raise(StandardTags::TagError.new("`host' tag requires the root page to have a `host' page part that contains the hostname."))
344     else
345       host
346     end
347   end

362   def site_area(tag)
363     page = tag.locals.page
364     unless page.part("site_area").nil?
365       page.part("site_area").content
366     else
367       case slug = page.url[1..-1].split(/\//).first
368       when nil
369         "homepage"
370       when /^\d/
371         "n#{slug}"

373         slug
374       end
